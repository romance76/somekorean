<template>
  <nav class="bg-white shadow-sm sticky top-0 z-50">
    <!-- Row 1: Logo + Search + User Actions -->
    <div class="max-w-[1200px] mx-auto px-4 flex items-center h-12 gap-3">
      <RouterLink to="/" class="flex-shrink-0">
        <img src="/images/logo.jpg" alt="SomeKorean" class="h-8" />
      </RouterLink>

      <!-- Search (desktop) -->
      <form @submit.prevent="goSearch" class="flex-1 max-w-xs hidden md:flex">
        <div class="relative w-full">
          <input v-model="searchQ" type="text" :placeholder="t('common.search') + '...'"
            class="w-full border border-gray-200 rounded-lg px-3 py-1.5 pr-8 text-xs focus:outline-none focus:ring-2 focus:ring-blue-400 transition"/>
          <button type="submit" class="absolute right-2.5 top-1/2 -translate-y-1/2 text-gray-400 hover:text-blue-500">
            <svg class="w-3.5 h-3.5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M21 21l-6-6m2-5a7 7 0 11-14 0 7 7 0 0114 0z"/>
            </svg>
          </button>
        </div>
      </form>

      <!-- Right Side -->
      <div class="ml-auto flex items-center gap-1.5 flex-shrink-0">
        <template v-if="auth.isLoggedIn">
          <!-- 알림 -->
          <RouterLink to="/notifications" class="relative p-1.5 text-gray-500 hover:text-blue-600">
            <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2"
                d="M15 17h5l-1.405-1.405A2.032 2.032 0 0118 14.158V11a6.002 6.002 0 00-4-5.659V5a2 2 0 10-4 0v.341C7.67 6.165 6 8.388 6 11v3.159c0 .538-.214 1.055-.595 1.436L4 17h5m6 0v1a3 3 0 11-6 0v-1m6 0H9"/>
            </svg>
            <span v-if="unreadNotifs > 0"
              class="absolute top-0 right-0 w-4 h-4 bg-red-500 text-white text-[10px] rounded-full flex items-center justify-center leading-none font-bold">
              {{ unreadNotifs > 9 ? '9+' : unreadNotifs }}
            </span>
          </RouterLink>
          <!-- 메시지 (sm+) -->
          <RouterLink to="/messages" class="p-1.5 text-gray-500 hover:text-blue-600 hidden sm:block">
            <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
              <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2"
                d="M8 10h.01M12 10h.01M16 10h.01M9 16H5a2 2 0 01-2-2V6a2 2 0 012-2h14a2 2 0 012 2v8a2 2 0 01-2 2h-5l-5 5v-5z"/>
            </svg>
          </RouterLink>
          <!-- 포인트 (md+) -->
          <RouterLink to="/points" class="hidden md:flex items-center gap-0.5 text-xs text-yellow-600 font-bold bg-yellow-50 px-2.5 py-1 rounded-lg hover:bg-yellow-100 transition">
            ⭐ {{ (auth.user?.points_total ?? 0).toLocaleString() }}P
          </RouterLink>
          <!-- 지갑 (md+) -->
          <button @click="showWalletPopup = !showWalletPopup" class="hidden md:flex items-center gap-0.5 text-xs text-yellow-700 font-bold bg-yellow-50 px-2.5 py-1 rounded-lg hover:bg-yellow-100 transition relative">
            🪙 {{ walletCoins.toLocaleString() }}
            <!-- Wallet dropdown -->
            <div v-if="showWalletPopup" class="absolute top-full right-0 mt-1 bg-white rounded-xl shadow-lg p-3 z-50 min-w-[180px] border border-gray-100" @click.stop>
              <div class="text-xs font-bold text-gray-500 mb-2">내 지갑</div>
              <div class="space-y-1.5">
                <div class="flex justify-between text-xs"><span>🪙 COIN</span><span class="font-bold text-yellow-600">{{ walletCoins.toLocaleString() }}</span></div>
                <div class="flex justify-between text-xs"><span>💎 GEM</span><span class="font-bold text-blue-600">{{ walletGems.toLocaleString() }}</span></div>
                <div class="flex justify-between text-xs"><span>⭐ STAR</span><span class="font-bold text-pink-500">{{ walletStars.toLocaleString() }}</span></div>
              </div>
              <RouterLink to="/games" class="block mt-2 text-xs text-center text-indigo-600 hover:text-indigo-700 font-semibold">게임 센터 →</RouterLink>
            </div>
          </button>
          <!-- 아바타 -->
          <RouterLink to="/dashboard"
            class="w-8 h-8 rounded-full overflow-hidden bg-blue-500 text-white flex items-center justify-center text-sm font-bold hover:ring-2 hover:ring-blue-300 flex-shrink-0 transition">
            <img v-if="auth.user?.avatar" :src="auth.user.avatar" class="w-full h-full object-cover" />
            <span v-else>{{ (auth.user?.name || auth.user?.username || '?')[0] }}</span>
          </RouterLink>
          <button @click="logout" class="text-xs text-gray-400 hover:text-red-400 hidden lg:block ml-1">{{ t('auth.logout') }}</button>
        </template>
        <template v-else>
          <RouterLink to="/auth/login" class="text-sm text-gray-600 hover:text-blue-600 px-3 py-1 hidden sm:block">{{ t('auth.login') }}</RouterLink>
          <RouterLink to="/auth/register" class="text-sm bg-blue-600 text-white px-3 py-1.5 rounded-lg hover:bg-blue-700 hidden sm:block">{{ t('auth.register') }}</RouterLink>
        </template>
        <!-- 언어 스위처 -->
        <button @click="toggleLang"
          class="flex items-center gap-1 text-xs font-bold px-2.5 py-1 rounded-lg border transition ml-1 flex-shrink-0"
          :class="langStore.locale === 'en' ? 'bg-blue-50 border-blue-300 text-blue-700' : 'bg-gray-50 border-gray-200 text-gray-600 hover:border-blue-300'">
          {{ langStore.locale === 'ko' ? '🇺🇸 EN' : '🇰🇷 한' }}
        </button>
        <!-- 햄버거 (mobile only) -->
        <button @click="mobileMenuOpen = true" class="md:hidden p-1.5 text-gray-600 hover:text-blue-600 ml-0.5">
          <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
            <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M4 6h16M4 12h16M4 18h16"/>
          </svg>
        </button>
      </div>
    </div>

    <!-- Row 2: Desktop nav — 관리자 메뉴 설정에 따라 동적 표시 -->
    <div class="border-t border-gray-100 bg-gray-50/80 hidden md:block">
      <div class="max-w-[1200px] mx-auto px-4 flex items-center h-9 overflow-x-auto scrollbar-hide gap-0.5">
        <RouterLink v-for="item in visibleDesktopNav" :key="item.key"
          :to="item.to" class="nav-link" :class="item.extraClass || ''">
          {{ item.label }}
        </RouterLink>
      </div>
    </div>

    <!-- Mobile Drawer (Teleport to body) -->
    <Teleport to="body">
      <Transition name="drawer-fade">
        <div v-if="mobileMenuOpen" class="fixed inset-0 z-[200] md:hidden" role="dialog">
          <!-- Backdrop -->
          <div class="absolute inset-0 bg-black/50 backdrop-blur-sm" @click="mobileMenuOpen = false"></div>
          <!-- Panel (slide from right) -->
          <Transition name="drawer-slide" appear>
            <div v-if="mobileMenuOpen" class="absolute right-0 top-0 h-full w-72 bg-white shadow-2xl flex flex-col">
              <!-- Header -->
              <div class="flex items-center justify-between px-4 h-12 bg-blue-600 text-white flex-shrink-0">
                <span class="font-bold text-sm">🇰🇷 SomeKorean</span>
                <button @click="mobileMenuOpen = false" class="p-1 text-white/80 hover:text-white rounded-lg">
                  <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                    <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M6 18L18 6M6 6l12 12"/>
                  </svg>
                </button>
              </div>

              <!-- User Section -->
              <div v-if="auth.isLoggedIn" class="px-4 py-3 bg-blue-50 border-b flex items-center gap-3 flex-shrink-0">
                <RouterLink @click="mobileMenuOpen = false" to="/dashboard"
                  class="w-10 h-10 rounded-full overflow-hidden bg-blue-500 text-white flex items-center justify-center text-sm font-bold flex-shrink-0 hover:ring-2 hover:ring-blue-300 transition">
                  <img v-if="auth.user?.avatar" :src="auth.user.avatar" class="w-full h-full object-cover" />
                  <span v-else>{{ (auth.user?.name || auth.user?.username || '?')[0] }}</span>
                </RouterLink>
                <div class="flex-1 min-w-0">
                  <div class="font-semibold text-gray-800 text-sm truncate">{{ auth.user?.name }}</div>
                  <div class="text-xs text-yellow-600">⭐ {{ (auth.user?.points_total ?? 0).toLocaleString() }}P · {{ auth.user?.level }}</div>
                </div>
              </div>
              <div v-else class="px-4 py-3 bg-blue-50 border-b flex gap-2 flex-shrink-0">
                <RouterLink @click="mobileMenuOpen = false" to="/auth/login"
                  class="flex-1 text-center py-2 border border-blue-600 text-blue-600 rounded-lg text-sm font-semibold hover:bg-blue-50 transition">
                  {{ t('auth.login') }}
                </RouterLink>
                <RouterLink @click="mobileMenuOpen = false" to="/auth/register"
                  class="flex-1 text-center py-2 bg-blue-600 text-white rounded-lg text-sm font-semibold hover:bg-blue-700 transition">
                  {{ t('auth.register') }}
                </RouterLink>
              </div>

              <!-- Mobile Search -->
              <form @submit.prevent="mobileSearch" class="px-4 py-2 border-b flex-shrink-0">
                <div class="relative">
                  <input v-model="searchQ" type="text" :placeholder="t('common.search') + '...'"
                    class="w-full border border-gray-200 rounded-lg px-3 py-2 pr-8 text-sm focus:outline-none focus:border-blue-400"/>
                  <button type="submit" class="absolute right-2.5 top-1/2 -translate-y-1/2 text-gray-400 hover:text-blue-500">
                    <svg class="w-4 h-4" fill="none" stroke="currentColor" viewBox="0 0 24 24">
                      <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M21 21l-6-6m2-5a7 7 0 11-14 0 7 7 0 0114 0z"/>
                    </svg>
                  </button>
                </div>
              </form>

              <!-- Nav Links — 관리자 메뉴 설정에 따라 동적 표시 -->
              <div class="flex-1 overflow-y-auto">
                <RouterLink v-for="item in visibleDrawerLinks" :key="item.to"
                  :to="item.to" @click="mobileMenuOpen = false"
                  class="flex items-center gap-3 px-4 py-3 hover:bg-blue-50 transition border-b border-gray-50 last:border-0"
                  :class="item.color || 'text-gray-700 hover:text-blue-600'">
                  <span class="text-lg w-6 text-center flex-shrink-0">{{ item.icon }}</span>
                  <span class="text-sm font-medium">{{ item.label }}</span>
                </RouterLink>
              </div>

              <!-- Bottom: Lang + Logout -->
              <div class="border-t px-4 py-3 flex items-center gap-2 flex-shrink-0 bg-gray-50">
                <button @click="toggleLang"
                  class="flex-1 py-2 text-xs font-bold rounded-lg border text-center transition"
                  :class="langStore.locale === 'en' ? 'bg-blue-50 border-blue-300 text-blue-700' : 'bg-white border-gray-200 text-gray-600 hover:border-blue-300'">
                  {{ langStore.locale === 'ko' ? '🇺🇸 English' : '🇰🇷 한국어' }}
                </button>
                <button v-if="auth.isLoggedIn" @click="logoutAndClose"
                  class="flex-1 py-2 text-xs font-bold rounded-lg border border-red-200 text-red-500 bg-red-50 hover:bg-red-100 transition">
                  {{ t('auth.logout') }}
                </button>
              </div>
            </div>
          </Transition>
        </div>
      </Transition>
    </Teleport>
  </nav>
</template>

<script setup>
import { ref, computed, onMounted } from 'vue'
import { useAuthStore } from '../stores/auth'
import { useLangStore } from '../stores/lang'
import { useSiteStore } from '../stores/site'
import { useRouter } from 'vue-router'
import axios from 'axios'

const auth         = useAuthStore()
const langStore    = useLangStore()
const siteStore    = useSiteStore()
const router       = useRouter()
const unreadNotifs = ref(0)
const searchQ      = ref('')
const mobileMenuOpen = ref(false)
const walletCoins  = ref(0)
const walletGems   = ref(0)
const walletStars  = ref(0)
const showWalletPopup = ref(false)

function t(key) { return langStore.$t(key) }

function toggleLang() {
  langStore.setLocale(langStore.locale === 'ko' ? 'en' : 'ko')
}

async function logout() {
  await auth.logout()
  router.push('/')
}

async function logoutAndClose() {
  mobileMenuOpen.value = false
  await auth.logout()
  router.push('/')
}

function goSearch() {
  if (!searchQ.value.trim()) return
  router.push({ name: 'search', query: { q: searchQ.value } })
  searchQ.value = ''
}

function mobileSearch() {
  if (!searchQ.value.trim()) return
  mobileMenuOpen.value = false
  router.push({ name: 'search', query: { q: searchQ.value } })
  searchQ.value = ''
}

// 전체 메뉴 목록 (key = 관리자 menu_settings의 key와 일치)
const allNavItems = computed(() => {
  const ko = langStore.locale === 'ko'
  return [
    { key: 'community',  to: '/community',  icon: '💬', label: t('nav.community') },
    { key: 'clubs',      to: '/clubs',      icon: '👥', label: ko ? '동호회' : 'Clubs' },
    { key: 'jobs',       to: '/jobs',       icon: '💼', label: t('nav.jobs') },
    { key: 'market',     to: '/market',     icon: '🛍️', label: t('nav.market') },
    { key: 'realestate', to: '/realestate', icon: '🏠', label: t('nav.realestate') },
    { key: 'directory',  to: '/directory',  icon: '🏪', label: t('nav.directory') },
    { key: 'chat',       to: '/chat',       icon: '💬', label: t('nav.chat') },
    { key: 'friends',    to: '/friends',    icon: '👫', label: t('nav.friends') },
    { key: 'match',      to: '/match',      icon: '💝', label: t('nav.match') },
    { key: 'games',      to: '/games',      icon: '🀄', label: t('nav.games') },
    { key: 'events',     to: '/events',     icon: '🎉', label: t('nav.events') },
    { key: 'ride',       to: '/ride',       icon: '🚗', label: t('nav.ride') },
    { key: 'elder',      to: '/elder',      icon: '💙', label: t('nav.elder'),   color: 'text-gray-700 hover:text-blue-600' },
    { key: 'news',       to: '/news',       icon: '📰', label: t('nav.news') },
    { key: 'qa',         to: '/qa',         icon: '❓', label: 'Q&A' },
    { key: 'recipes',    to: '/recipes',    icon: '🍳', label: '레시피' },
    { key: 'shorts',     to: '/shorts',     icon: '📱', label: t('nav.shorts'),   color: 'text-red-600 hover:text-red-700',   extraClass: 'font-semibold text-red-500' },
    { key: 'shopping',   to: '/shopping',   icon: '🛒', label: t('nav.shopping'), color: 'text-green-700 hover:text-green-800', extraClass: 'font-semibold text-green-600' },
    { key: 'groupbuy',   to: '/groupbuy',   icon: '🤝', label: t('nav.groupbuy'), color: 'text-orange-600 hover:text-orange-700', extraClass: 'font-semibold text-orange-500' },
    { key: 'mentor',     to: '/mentor',     icon: '🎓', label: t('nav.mentor'),   color: 'text-teal-600 hover:text-teal-700',   extraClass: 'font-semibold text-teal-600' },
    { key: 'ai',         to: '/ai',         icon: '🤖', label: t('nav.aiSearch'), color: 'text-violet-600 hover:text-violet-700', extraClass: 'font-semibold text-violet-600' },
    { key: 'points',     to: '/points',     icon: '⭐', label: ko ? '포인트' : 'Points' },
    { key: 'messages',   to: '/messages',   icon: '✉️', label: ko ? '메시지' : 'Messages' },
  ]
})

// 관리자 설정에서 enabled인 메뉴만 필터
const visibleNavItems = computed(() =>
  allNavItems.value.filter(item => siteStore.isEnabled(item.key))
)

// 데스크탑 네비 (points, messages는 상단 아이콘으로 표시하므로 제외)
const visibleDesktopNav = computed(() =>
  visibleNavItems.value.filter(item => !['points','messages'].includes(item.key))
)

// 모바일 드로어
const visibleDrawerLinks = visibleNavItems

onMounted(async () => {
  if (auth.isLoggedIn) {
    try {
      const { data } = await axios.get('/api/notifications/unread')
      unreadNotifs.value = data.count
    } catch { }
    try {
      const { data } = await axios.get('/api/wallet/balance')
      walletCoins.value = data.coin ?? 0
      walletGems.value  = data.gem  ?? 0
      walletStars.value = data.star  ?? 0
    } catch { }
  }
})
</script>

.nav-link {
  @apply text-xs text-gray-600 hover:text-blue-600 px-2.5 py-1.5 rounded-md hover:bg-blue-50 transition font-semibold whitespace-nowrap;

  @apply text-blue-600 bg-blue-50;
}
.scrollbar-hide::-webkit-scrollbar { display: none; }
